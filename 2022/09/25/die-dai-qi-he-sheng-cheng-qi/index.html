迭代器和生成器

javascript
发布日期:   2022-09-25
文章字数:   1.2k

迭代器和生成器

迭代器的特征

  • let iter = arr[Symbol.iterator]();创建迭代器（可迭代对象）
  • iter.next()的值为{value: 10, done: false}value 就是值，done 表示迭代结束了没有

迭代和遍历的区别？

  • 【过程】迭代强调的依次取出，不能确定可以取出的值有多少，也不能保证去把数据全部取
  • 【结果】遍历必须保证数据的长度，循环不断的全部取出，针对数据量过大的情况下使用遍历，需要事件过长

可迭代对象

  • 数组可迭代对象let arr = [10,20,30];
  • 对象不是可迭代对象，因为对象是无序的let obj = {name:"javascript",age:27};
  • 字符串可迭代对象 let str = "hello"
  • map对象也是可迭代对象let map1 = new Map([["k1","v1"],["k2","v2"],["k3","v3"]]);
  • 数据集合也是可迭代对象let btns = document.querySelectorAll("button");

可迭代对象，可以用 for…of 来循环

for(let val of btns){
    console.log(val);
}

设计一个可迭代对象

let classes = {     // 原本不是一个可迭代的对象
    stus:["小明","小红","小刚","小黄"],
    [Symbol.iterator](){
        let index = 0;  // 对应的索引
        const this_ = this;
        return {
            next(){
                // if(还有元素)
                // if(这个元素的索引<stus长度)
                if(index<this_.stus.length){
                    const ret = {value: this_.stus[index], done: false};
                    index++;
                    return ret;
                }else{
                    return {value: undefined, done: true};
                }
            }
        }
    }
}

for(let val of classes){
    console.log(val);
}

生成器函数

生成器函数的执行机制总结：

  1. 通过调用生成器函数来创建迭代器 (此时生成器的代码不会执行)
  2. 通过迭代器调用next执行生成器函数里面的代码
  3. 执行到 yieId ，暂时跳出函数，并把yieId后面的值(加工成对象，返回到next的地方)

一步一步生成数据，可以创建迭代器

function* genFunc() {   // 第一个不同：有一个 *
    console.log("执行了genFunc");
    yield 1;    // 第一个不同：关建字 yield
    console.log("执行了genFunc2");
    yield 2;
    console.log("执行了genFunc3");
    yield 3;
}

生成器函数的执行机制

生成器函数() 并不是执行函数里面的代码,而是创建了一个迭代器

let iter = genFunc();   // 创建一个迭代器

{value: 1, done: false} 第一次调用next,才开始去执行生成器函数 genFunc 里面的代码,执行遇到 yieId 的时候就停下来，并且跳出这个函数，返回 yieId 后面的值(加工成一个对象)给调用 .next() 的地方

console.log(iter.next());   // {value: 1, done: false}

再次调用next()，并且进入到生成器函数genFunc里面执行代码

console.log(iter.next());

for(let val of iter){
    console.log(val);
}

生成器的应用场景

应用场景：第一个异步请求成功后再去发送第二个请求，第二个请求发送后，再去发送第三个请求。(每一个请求是基于前面一个请求成功之后，才发送请求，因为要拿到上一次的数据)

function XxxxAPI(x){
    return new Promise((resolve,reject)=>{
        setTimeout(() => {
            resolve(++x)
        }, 1000);
    })
}

基于 Promise 的写法

XxxxAPI(0).then(resolve1=>{
    console.log(`第1次请求结果为：${resolve1}`);
    return XxxxAPI(resolve1);
}).then(resolve2=>{
    console.log(`第2次请求结果为：${resolve2}`);
    return XxxxAPI(resolve2);
}).then(resolve3=>{
    console.log(`第3次请求结果为：${resolve3}`);
    return XxxxAPI(resolve3);
})

生成器创建迭代器的写法

function* genFunc(x){
    let result1 = yield XxxxAPI(x);
    console.log(`第一次请求结果为：${result1}`);
    let result2 = yield XxxxAPI(result1);
    console.log(`第一次请求结果为：${result2}`);
    let result3 = yield XxxxAPI(result2);
    console.log(`第一次请求结果为：${result3}`);
}

let iter = genFunc(0);

iter.next().value   // 拿到Promise对象
iter.next().value.then(result1=>{
    return iter.next(result1).value
}).then(result2=>{
    return iter.next(result2).value
}).then(result3=>{
    return iter.next(result3)
})

​ 上面 function* genFunc(x){} 格式很像 async + await 的格式。只是长得像，还是有区别的；

​ 生成器函数调用不会执行，调用会创建迭代器，需要你next()才能执行，而 async + await 是后来才出现的语法糖 

async function genFunc(x){
    let result1 = await XxxxAPI(x);
    console.log(`第一次请求结果为：${result1}`);
    let result2 = await XxxxAPI(result1);
    console.log(`第一次请求结果为：${result2}`);
    let result3 = await XxxxAPI(result2);
    console.log(`第一次请求结果为：${result3}`);
}

封装自己的迭代器

function XxxxAPI(x){
    return new Promise((resolve,reject)=>{
        setTimeout(() => {
            resolve(++x)
        }, 1000);
    })
}

// iter.next().value.then(result1=>{
//      iter.next(result1).value.then(result2=>{
//      iter.next(result2).value.then(result3=>{
//      iter.next(result3)
// })
// })
// })

// asyncFunc 是我们封装的函数
function asyncFunc(genFn,...params){
    let iter = genFn(...params);
    const next = x =>{
        // 如果没有下一个值，就return
        let {value,done} = iter.next(x);
        if(done)return;
        // 如果有值
        value.then(next)
    }
    next()
}


// 简单模拟了一下 async + await 的原理
asyncFunc(function* genFunc(x){
    let result1 = yield XxxxAPI(x);
    console.log(`第一次请求结果为：${result1}`);
    let result2 = yield XxxxAPI(result1);
    console.log(`第一次请求结果为：${result2}`);
    let result3 = yield XxxxAPI(result2);
    console.log(`第一次请求结果为：${result3}`);
},0)
