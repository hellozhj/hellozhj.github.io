一. uView

官网：https://v1.uviewui.com/

uView是uni-app生态专用的UI框架，uni-app 是一个使用 Vue.js 开发所有前端应用的框架，开发者编写一套代码， 可发布到iOS、Android、H5、以及各种小程序(微信/支付宝/百度/头条/QQ/钉钉)等多个平台

五. 网络请求

uni-app使用uni.request来发送网络请求：

uni.request({
  url: 'http://kumanxuan1.f3322.net:8001/index/index',
  success: (res) => {
    console.log(res);
  }
});

封装请求：

在根目录新建request文件夹，用来存放和请求相关的文件。

function request({
  url = "",
  data = {},
  timeout = 10000,
  method = "GET",
  dataType = "json",
  successCb = null,
  failCb = null,
} = {}) {
  let _url = url
  if (_url.startsWith('/')) {
    _url = _url.substring(1)
  }
  uni.request({
    url: `http://kumanxuan1.f3322.net:8001/${_url}`, // 路径
    data, // 请求的参数
    header: {
      // "content-type"用于定义网络文件的类型和网页的编码，决定浏览器将以什么形式、什么编码读取这个文件
      "content-type": "application/json",
    },
    timeout, // 请求超时时间
    method, // HTTP 请求方法
    dataType, // 返回的数据格式
    success: (res) => {
      // 请求成功回调函数
      console.log(res);
      const {
        data,
        errmsg,
        errno
      } = res.data;
      if (res.statusCode === 200 && errno === 0) {
        successCb && successCb(data);
      } else {
        uni.showToast({
          title: errmsg || "错误！",
          icon: "none",
          duration: 1500,
        });
        failCb && failCb();
      }
    },
    fail: (err) => {
      // 请求失败回调函数
      uni.showToast({
        title: err.errMsg || "错误！",
        icon: "none",
        duration: 1500,
      });
      failCb && failCb();
    },
  });
}

export default request;

在request文件夹下新建api.js存放每个接口地址，这样做的好处是如果后端改接口名，而我们项目中有多处调用到改名的接口，那么我们只需要更改这份文件即可：

const apiIndex = "index/index"; // 首页接口

export { apiIndex };

首页调用接口：

<!-- pages/index/index.vue -->
<script>
import request from "../../request/request.js";
import { apiIndex } from "../../request/api.js";

export default {

        console.log(res)
      },
    });

export function checkToken({
	cbIfHasToken = null, // 是否有token
	pathForGoingBack = '', // 往回退的页面路径
	isTabBarPage = true // 往回退的页面是否有 tabBar 页面
} = {}) {
	const token = uni.getStorageSync('token');
	if (token) {
		// 有 token 执行回调
		cbIfHasToken()
	} else {
		// 没有就提示跳转用户注册界面	
		uni.switchTab({
			url: '/page/user/user',
			success: () => {
				uni.setStorageSync('pathForGoingBack', pathForGoingBack)
				uni.setStorageSync('isTabBarPage', isTabBarPage)

				// 需要延迟提示
				setTimeout(() => {
					uni.showToast({
						title: '请先登录',
						icon: 'none'
					})
				}, 200)
			}
		})
	}
}

export function goBackPath() {
	const pathForGoingBack = uni.getStorageSync('pathForGoingBack');
	const isTabBarPage = uni.getStorageSync('isTabBarPage');
	// 如果缓存里面有往回跳的路径
	if (pathForGoingBack) {
		if (isTabBarPage) {
			uni.switchTab({
				url: pathForGoingBack,
				success: () => {
					uni.removeStorageSync('pathForGoingBack');
					uni.removeStorageSync('isTabBarPage');
				}
			})
		} else {
			// 普通页面
			// 不能用 navigateBack,因为跳转到 user 页面的时候，页面栈已经清空只剩下 user 页面

			// redirectTo 返回到 pathForGoingBack 对应页面后，再返回会跳转到 pathForGoingBack 的上一页，如果是 navigateTo , 跳转到 pathForGoingBack 对应页面后，再返回会跳转到登录页面

		}
	}
}
