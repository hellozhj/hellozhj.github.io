NodeJs - 总结

nvm命令

  • nvm ls 查看当前的电脑上下载了哪些 node 版本

  • nvm install 下载指定版本的node

  • nvm use 切换指定的node版本

  • nvm version 查看当前安装 nvm 的版本

  • nvm root 查看nvm的安装目录

  • nvm uninstall 卸载

  • node -v 查看当前使用 node 的版本

  • node 用于执行js代码

    node js文件的路径 —> node ./a/b/c/xx.js

  • 环境变量

    快速连接到一个执行程序(.exe/.cmd)一种功能

  • 知识点

    * 服务器的工作原理

    * 1. 浏览器根据ip找到服务器的主机

    * 2. 根据端口找到服务器软件

    * 3. 判断请求的类型(静态/动态)

    * 静态 —> 读取文件 —> 响应

    * 动态 —> 操作数据库.. —> 响应

使用http实现一个极其简单的服务器

  1. 引入http 模块
  2. 创建一个服务器对象
  3. 监听端口和ip
  4. 监听浏览器请求服务器事件
  5. 响应数据回浏览器
// 1， 引入http 模块
const http = require('http'); // 这个写法是node里面比较特殊的写法，目前大家就习惯一下就好
// 2， 创建一个服务器对象
const server = http.createServer();
// 3， 监听端口和ip
server.listen(8080,'192.168.12.131');
// 4， 监听浏览器请求服务器事件
server.on('request',(req,res)=>{
    // 5， 响应数据回浏览器
    res.end("<h1>helloworld</h1>")
})

统一处理静态资源

  • 目标： 希望把所有的静态都能统一管理，而不是每个静态都要判断
  • 思路： url 就是 完整的 要读取的资源的路径
  • url : /views/index.html
  • 读取的时候 __dirname + /views/index.html
  • 就可以使用 url 代替 可以变化的路径部分
const http = require("http");
const fs = require("fs");
const path = require("path");
const server = http.createServer();
server.listen(8080, "192.168.12.131");
server.on("request", (req, res) => {

  fs.readFile(path.join(__dirname, req.url), (err, data) => {
    res.end(data);
  });
});

判断是否静态资源

const http = require("http");
const fs = require("fs");
const path = require("path");
const server = http.createServer();
server.listen(8080, "192.168.12.131");
server.on("request", (req, res) => {
  // 目标： 希望可以把静态请求和动态请求分开
  // 思路： 把静态资源放在一些固定的 文件夹里面   views /  assets
  // 此时当用户请求静态资源的时候，一定是以这些文件夹开头
  // 只要判断， 请求 是否是 以 指定的文件夹开头就行
  // if(req.url 是否 以 /views 开头或者以 /assets 开头)
  if (/^\/views/.test(req.url) || /^\/assets/.test(req.url)) {
    fs.readFile(path.join(__dirname, req.url), (err, data) => {
      res.end(data);
    });
  }
});

响应动态数据

/**
 *  目标： 响应一些动态数据回到浏览器
 *
 */
const http = require("http");
const fs = require("fs");
const path = require("path");
const server = http.createServer();
server.listen(8080, "192.168.12.131");
server.on("request", (req, res) => {
  //  服务器接收到请求，除了静态资源，就是动态请求
  // 动态请求有： 登录 / 注册 / 文章列表 / 分布请求  / 评论 / 评论列表 ...
  //   登录的逻辑：  ,  用户名和密码是否正确 ..
  //   注册的逻辑 ： 判断用户名是否合法 ， 用户名是否已经存在 ...
  //   获取文章列表 ： 获取哪种文章 ， 到数据库里面根据一定的条件获取
  //   获取文章详情 ： 根据id到数据库里找
  //   ...
  //   动态请求每个请求都需要单独处理 ， 只能一个一个判断
  if (req.url === "/login") {
    //   一般在后端会响应回来一个 json 格式的对象
    // 里面包含： code - 业务代码 , 不同的数字代表不同的状态 ， msg -- 业务消息 ， data --- 响应回来的数据
    res.end(
      JSON.stringify({
        code: 200,
        msg: "ok",
        data: {
          token: "531ga654fg35adfg4a3dfg1as35df1",
        },
      })
    );
  }
  if (req.url === "/register") {
    res.end("用户名不合法");
  }
  //   ...
});

server

// 当前这个 05-server.mjs 就负责开启服务器
import http from "http";
// 导入 处理静态资源的函数
import staticAssets from "./05-static.mjs";
// 导入 处理动态请求的 函数
import { dynamic } from "./05-dynamic.mjs";
const server = http.createServer();
// http协议的默认端口是 80 , https  协议的端口默认是443
server.listen(8080);

server.on("request", (req, res) => {
  // 处理静态
  staticAssets(req, res);
  //   处理动态请求
  dynamic(req, res);
});

static

/**
 *  这个文件负责实现静态资源处理
 */

import fs from "fs";
import path from "path";
import url from "url";

const __dirname = path.dirname(url.fileURLToPath(import.meta.url));

// 静态处理的思路： 判断 请求 以 什么开头
export default function staticAssets(req, res) {
  if (/^\/assets|\/views/.test(req.url)) {
    // 读取静态文件
    fs.readFile(path.join(__dirname, req.url), (err, data) => {
      res.end(data);
    });
  }
}

  // simple query
  // 执行sql查询
  connection.query("SELECT * FROM `student`", function (err, results, fields) {
    console.log(results); // results contains rows returned by server - 执行结果
    console.log(fields); // fields contains extra meta data about results, if available - 查询的时候得到的字段
    res.send({
      code: 200,
      msg: "ok",
      data: results,
    });
  });
});

跨域

  • 跨过域名请求数据，进行数据交互
  • 浏览器默认是不认同，它认为是危险的
  • 现代的web开发，一定会在开发的时候产生跨域
  • 只要 协议名 / ip / 端口 三者中有一个不同，就会产生跨域
  • aaa.com 网站， 向 bbb.com 的服务器发起了请求 —————— > 跨域

jsonp

Step1: 创建 callback 方法

Step2: 插入 script 标签

Step3: 后台接受到请求，解析前端传过去的 callback 方法，返回该方法的调用，并且数据作为参数传入该方法

Step4: 前端执行服务端返回的方法调用

代理服务器

proxy

/**
 *  代理服务器
 *
 */
const express = require("express");
const cors = require("cors");
const http = require("http");
const app = express();
app.listen(8080);
app.use(cors());

app.get("/a", (req, res) => {
  // 当c.html 发一个请求过来
  // 我们把这个请求转发到 data-server
  http.get("http://127.0.0.1:8081/b", (response) => {
    let data = "";
    response.on("data", (chuck) => {
      data += chuck;
    });
    response.on("end", () => {
      //   到了这里是 http 向 8081 请求完成
      console.log(data);
      //   再把从 8081 提到的数据响应回 浏览器
      res.send(data);
    });
  });
});
