购物车【全选/多选框，计算属性get,set方法】

方法1：(多选框数组)

<body>
    <div id="app">
        <h4>购物车</h4>
        <table id="table">
            <thead>
                <tr>
                    <th>
                        <input type="checkbox" id="btn1" v-model="isCheckedAll" />全选/全不选
                    </th>
                    <th>书籍名称</th>
                    <th>出版日期</th>
                    <th>价格</th>
                    <th>数量</th>
                </tr>
            </thead>
            <tbody id="tbody">
                <tr v-for="item,index in books">
                    <td><input type="checkbox" v-model="checkList[index]" />
                    </td>
                    <td>{{item.name}}</td>
                    <td>{{item.date}}</td>
                    <td>{{item.price}}</td>
                    <td>
                        <button @click="stepFn(-1,index)">-</button>
                        <span>{{item.count}}</span>
                        <button @click="stepFn(1,index)">+</button>
                    </td>
                    <td><button @click="del(index)">移除</button></td>
                </tr>

                <tr>
                    <td colspan="5" class="totalc">
                        总价: <span id="total">￥{{total}}元</span>
                    </td>
                </tr>
            </tbody>
        </table>
    </div>
    <script>
        /*
         1. 渲染数据  
         2. 计算总价
         3. 实现步进器功能
         4. 实现全选,全不选的功能  方式一 使用数组来存储选中状态
         */

        new Vue({
            el: '#app',
            data: {
                // 每一项多选框的选中状态 [false,false,false,false]，可以用数组控制【重】
                checkList: [],
                books: [
                    { name: "算法导论", date: "2007-9", price: 100, count: 1 },
                    { name: "UNIX编程艺术", date: "2027-9", price: 120, count: 3 },
                    { name: "Vue程序设计", date: "2017-9", price: 60, count: 1 },
                    { name: "颈椎康复", date: "2009-9", price: 30, count: 2 },
                ],
            },
            methods: {
                del(i) {
                    this.books.splice(i, 1);
                    //同步删除下复选框数组，不能只加不减【重】
                    this.checkList.splice(i, 1);
                },
                // 步进器，两个功能结构大致相同，作用于同一个数据，可以用同一个方法表示【扩展】
                stepFn(step, i) {
                    if (step == -1 && this.books[i].count == 1) {
                        return;
                    }
                    this.books[i].count += step;
                }
            },
            computed: {
                total() {
                    // 利用reduce第三个值index，该位置的数组是否为true，选中就计算。 【扩】
                    return this.books.reduce((pre, cur, index) => pre + (this.checkList[index] ? cur.price * cur.count : 0), 0);
                },
                isCheckedAll: {
                    // 计算属性的get(),set() 格式 【扩，重】
                    get() {
                        // 复选框都没有了，全选框不能勾选上【细】
                        if (this.checkList.length == 0) {
                            return false;
                        }
                        //判断checkList数组中是否有一个为false,如果是 则不选中,如果没有 则选中【扩,重】
                        // 如果this.checkList.includes(false)  返回true --- 设置不选中  要取反
                        return !this.checkList.includes(false);
                    },
                    // 当用户手动修改isCheckedAll的值后就会触发set()方法
                    set(newVal) {
                        // newVal 是收到用户手动修改isCheckedAll的值
                        this.checkList = this.books.map(() => newVal);
                    }
                }
            },
            // 每一项多选框的选中状态，默认长度是数据的长度【重】
            created() {
                this.checkList = this.books.map(item => false);
            }
        })
    </script>
</body>

        new Vue({
            el: '#app',
            data: {
                books: [
                    { name: "算法导论", date: "2007-9", price: 100, count: 1 },
                    { name: "UNIX编程艺术", date: "2027-9", price: 120, count: 3 },
                    { name: "Vue程序设计", date: "2017-9", price: 60, count: 1 },
                    { name: "颈椎康复", date: "2009-9", price: 30, count: 2 },
                ],
            },
            methods: {
                del(i) {
                    this.books.splice(i, 1);
                },
                stepFn(step, i) {
                    if (step == -1 && this.books[i].count == 1) {
                        return
                    }
                    this.books[i].count += step;
                }
            },
            computed: {
                total() {
                    return this.books.reduce((pre, cur) => pre + (cur.isChecked ? cur.price * cur.count : 0), 0)
                },
                // 另一种求和写法：【扩】
                //if (cur.isShow) {
                //    return pre + cur.price * cur.count;
                //} else {
                //return pre;
                //}
                isCheckedAll: {
                    get() {
                        // 默认全选为 true  【开关思想】
                        let flag = true;
                        // 列表为空，全选为空
                        if (this.books.length == 0) {
                            return;
                        }
                        for (var i = 0; i < this.books.length; i++) {
                            // 判断是否有一项为 false
                            if (!this.books[i].isChecked) {
                                // flag = false;
                                return;
                            }
                        }
                        return flag;
                    },
                    set(newVal) {
                        this.books.forEach(item => {
                            item.isChecked = newVal;
                        })
                    }
                }

            },
            // 给每一个对象增加一个是否选中的属性,isChecked:false 【扩】

                // 在Vue2.0中，单独修改数组的某个元素的时候，触发不了视图层的修改。
                // this.books.forEach(item => {
                //     item.isChecked = true;
                // });

                // this.$set(要修改的数组,要修改的下标,要修改成的值)
                this.books.forEach(item => {
                    this.$set(item, "isChecked", false);
                });
            }
        })
    </script>

tabbar 【插槽、组件】

1.子组件添加类名，值是点击的组件添加类名，值可以是计算属性。

2.子组件定义标签的姓名属性，和被点到的属性 标题属性

3.父组件给每一个组件姓名，和被点击的值，由父组件控制 ，子组件显示父组件给的标题

4.点击不同的子组件,拿到不同的子组件的iname,修改父组件控制的fcall

5.每个组件变的是图片，可以用插件

// tabbar 组件

纯添加数组的方法

<body>
    <div id='app'>
        <h1>任务列表</h1>
        <ul>
            <li v-for="item,index in task" :class="{delTxt:isChecked[index]?true:false}" :key="index">
                <input type="checkbox" v-model="isChecked[index]"><span @click="changeInp(index)"
                                                                        v-show="isShow[index]">{{item}}</span>
                <input type="text" v-show="isInp[index]" @blur="blurInp(index)" v-model="upDate">
            </li>
        </ul>
        <p>
            任务总数{{task.length}};还有{{surplus}}未完成；
            【<a href="javascript:;" @click="del">完成</a>】
        </p>
        <p>
            <input type="text" placeholder="请输入任务名称" class="inp" v-model="txtVal">
            <button @click="add">添加</button>
        </p>
    </div>
    <script>
        new Vue({
            el: '#app',
            data: {
                task: ["预习Vuex知识点", "编写课后联系题", "抄写单词20遍"],
                txtVal: "",
                isChecked: [],
                isShow: [],
                isInp: [],
                upDate: "",
            },
            methods: {
                add() {
                    if (!this.txtVal) {
                        alert("请输入");
                        return;
                    }
                    this.task.push(this.txtVal);
                    this.isChecked.push(false);
                    this.isShow.push(true);
                    this.isInp.push(false);
                    this.txtVal = "";
                },
                del() {
                    for (var i = 0; i < this.isChecked.length; i++) {
                        if (this.isChecked[i]) {
                            this.task.splice(i, 1);
                            this.isChecked.splice(i, 1);
                            this.isShow.splice(i, 1);
                            this.isInp.splice(i, 1);
                            i--;
                        }
                    }
                },
                changeInp(i) {
                    if (this.isChecked[i]) {
                        return
                    }
                    // 单独修改数组的某个元素的时候，触发不了视图层的修改 $set()方法来修改元素
                    // this.isShow[i] = false;
                    this.$set(this.isShow, i, false);
                    // this.isInp[i] = true;
                    this.$set(this.isInp, i, true);
                },
                blurInp(i) {
                    if (!this.upDate) {
                        this.$set(this.isInp, i, false);
                        this.$set(this.isShow, i, true);
                        return
                    }
                    this.task[i] = this.upDate;
                    this.$set(this.isInp, i, false);
                    this.$set(this.isShow, i, true);
                    this.upDate = "";
                }
            },
            computed: {
                surplus() {
                    let ret = this.isChecked.filter(item => item == false);
                    return ret.length;
                },

            },
            created() {
                this.isChecked = this.task.map(item => false);
                this.isShow = this.task.map(item => true);
                this.isInp = this.task.map(item => false);
            }
        })
    </script>

粗略版

<body>
    <div id='app'>
        <h2>任务列表</h1>
        <div class="list" v-for="item,index in task" :key="index">
            <input type="checkbox" v-model="isCheck[index]">
            <span :class="{deltxt:flag(index)}" v-show="item.isShow==1"
                  @click="changeInp(index)">{{item.title}}</span>
            <input type="text" v-show="item.isShow==2" v-model="upDateTxtVal" @blur="change(index)">
        </div>
        <p>
            任务总数{{total}};还有{{surplus}}未完成；
            【<a href="javascript:;" @click="del">完成</a>】
        </p>
        <p class="btn">
            <input class="txt" type="text" placeholder="请输入任务名称" v-model="txtVal">
            <button @click="add">添加</button>
        </p>
    </div>
    <script>
        new Vue({
            el: '#app',
            data: {
                // isshow: true,
                txtflag: 1,
                upDateTxtVal: "",
                isCheck: [false, false, false],
                txtVal: "",
                task: [
                    { title: "预习Vuex知识点", isShow: 1 },
                    { title: "编写课后练习题", isShow: 1 },
                    { title: "抄写单词20遍", isShow: 1 }
                ],
            },
            methods: {
                add() {
                    if (!this.txtVal.trim()) {
                        alert("请输入任务！！");
                        this.txtVal = "";
                        return;
                    }
                    this.task.push({ title: this.txtVal, isShow: 1 });
                    this.isCheck.push(false)
                    this.txtVal = "";
                },
                del() {
                    for (i = 0; i < this.isCheck.length; i++) {
                        if (this.isCheck[i] == true) {
                            this.task.splice(i, 1);
                            this.isCheck.splice(i, 1);
                            // 循环删除数组，每一项会向前，索引要减一
                            i--;
                        }
                    }
                },
                changeInp(i) {
                    if (this.isCheck[i]) {
                        return
                    }
                    this.task[i].isShow = 2;
                },
                change(i) {
                    this.task[i].title = this.upDateTxtVal;
                    this.task[i].isShow = 1;
                    this.upDateTxtVal = "";
                }
            },
            computed: {
                total() {
                    return this.task.length;
                },
                surplus() {
                    let ret = this.isCheck.filter(item => {
                        return item == false;
                    })
                    return ret.length;
                },
                flag() {
                    return function (i) {
                        return this.isCheck[i];
                    }
                },
            }
        })
    </script>
</body>

                    // item.is_show = item.title.includes(this.txtVal);

                    // 第二种
                    item.is_show = item.title.indexOf(this.txtVal) == -1 ? false : true;
                });
            },
            delFn(val) {
                this.info.splice(val, 1);
            },
            inp() {
                if (!this.txtVal) {
                    this.info.forEach((item) => {
                        item.is_show = true;
                    });
                }
            },
        },
    };
</script>
