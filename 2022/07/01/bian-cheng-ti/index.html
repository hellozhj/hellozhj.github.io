function getArr(min, max, total) {
    // 因为需要返回数组，先声明一个数组

        // 取随机数推入空数组

function reStr(cStr) {

res.menus.forEach((el) => {
    if (el.hidden === 0) {
        if (el.level === 0) {
            // 父亲
            result.push({
                ...el,
                children: [],
            });
        } else {
            // 排序的作用：其实跑到这个逻辑的时候，result里面已经把父亲都添加完了

            // 孩子
            let parentIndex = result.findIndex((item) => {
                // 父级的id等于孩子的parentId
                return item.id === el.parentId;
            });
            if (parentIndex > -1) {
                result[parentIndex].children.push(el);
            }
        }
    }
});

// parentId 为-1为第一节点 ,其他的 都为parentId   所在数据的子类  
function getTreeData(list){
    // 最后的结果
    var treeData=[];
    var map={};
    list.forEach( (item) =>{
        item['name']=item.classifierName;
        // {root: {id: 'root', classifierName: 'root', level: '0', orderNumber: '1', parentId: '-1', …},}
        map[item.id]=item;
    })
    // console.log(map);
    list.forEach( (item)=> {
        // 每一项的parentId，是否存在对应map对象的键名
        var parent = map[item.parentId];
        // 没有没有对应的parentId,说明是顶级的
        if (parent) {
            // 有的给到children数组
            (parent.children || ( parent.children = [] )).push(item);
        } else {
            treeData.push(item);
        }
    })
    return treeData;
}
