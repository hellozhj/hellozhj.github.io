排他思想

思路：点击哪一个，就只有那一个生效，其他的不生效。在事件执行函数里面，在该点击生效之前，先清除所有的元素的效果

需求：给你5个按钮，按下其中一个，单改变它的背景颜色，其他按钮颜色不变。

eg：

var btn = document.querySelectorAll("button");
// 外层循环只是为了给每个按钮绑定点击事件
for (var i = 0; i < btn.length; i++) {
    btn[i].onclick = function () {
        // 思考怎么去掉其他的
        // 先全部去掉所有按钮的背景色
        // 里面的循环为了每次点击一个按钮之后先全部去掉背景色
        for (var j = 0; j < btn.length; j++) {
            btn[j].style.background = "";
        }
        // 再给用户点的那个加上背景色
        this.style.background = "tan";
    }
}

里面的for循环必须发生在 点击按钮的事件中，给按钮加上背景色之前，去掉每一个按钮的背景色

var box = document.querySelector("#box");

var prev = null;
box.onclick = function (e) {
    // 如果点击的目标元素不是box这个对象
    if (e.target !== box) {
        // 如果存在上一次点击的元素
        if (prev) {
            // 就去掉背景色
            prev.style.background = "";
        }
        // 再给目标元素添加背景色
        e.target.style.background = "tan";
        // 添加完之后把这个目标元素设置为上一次点击的元素
        prev = e.target;
    }
}

数组(或者字符串中)出现次数最多 出现了多少次

  1. 先把数组转化成字符串，str[下标]可以得到每一个字
  2. 新建一个空的对象来储存每一个字的属性，和每个字出现次数的值
  3. 判断新建的这个对象里面，有没有每个字str[i]
  4. 如果没有就设置这个属性的值为出现了一次，如果本来就有就+1次 ++
var str = students.join("");
var obj = {};

for (var i = 0; i < str.length; i++) {
    // char是每个字符
    // obj[char]是这个对象有没有这个字符属性
    var char = str[i];
    if (obj[char]) {
        // 有这个属性就加一次
        obj[char]++;
    } else {
        // 没有就把这个属性赋值为一次
        obj[char] = 1;
    }
}
// 输出这个保存了每个字和出现次数的对象
console.log(obj);

// 先设置一个最大值的初始值
var max = 0;
// 新建一个变量来储存出现最多次数的字符
var maxchar = "";

for (var key in obj) {
    if (obj[key] > max) {
        max = obj[key];
        maxchar = key;
    }
}
console.log(`出现次数最多的字符是${maxchar},出现的次数数${max}次`);

    // 非空校验

        var rows = +row.value;  //行数
        var cols = +col.value;  //列数
        var table = document.createElement("table");
        // 外层循环控制行数tr
        for (var i = 1; i <= rows; i++) {
            var tr = document.createElement("tr");
            // 里层循环控制列数td
            for (var j = 1; j <= cols; j++) {
                var td = document.createElement("td");
                td.innerHTML = `第${i}行,第${j}列`;
                // 把td插入tr里面
                tr.appendChild(td);
            }
            // 把tr插入table后面
            table.appendChild(tr);
        }
        // 把表格插入body后面
        document.body.appendChild(table);
        // 清空输入框的值
        row.value = "";
        col.value = "";
    } else {
        alert("请输入内容")
    }
}

事件的三个阶段（冒泡）

  1. 第一阶段是从外向内找到目标元素**(捕获阶段)**

  2. 第二阶段执行了事件处理程序**(目标阶段)**

  3. 第三阶段 执行事件的时候,事件会从内向外传递**(冒泡阶段)**

    如果祖先标签也绑定了同类事件,就会一层一层从里往外触发

  4. window > document > html > body
