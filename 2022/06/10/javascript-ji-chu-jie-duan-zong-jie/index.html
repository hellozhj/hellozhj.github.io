自增与自减

自增

i++; //i++ 相当于i = i + 1 (先计算，后自增)
++i; //++i 相当于i = i + 1 (先自增，后计算)

两者的不同点在于：参与运算的过程中，++在前面是先自增，在计算，++在后面先计算后自增

如果一行代码只有i++或者++i，其效果都是i+1。

var i = 1;
i++;
console.log(i);  //2
++i;
console.log(i);  //3

一旦参与了其他运算，二者就有区别了

i++

var i =1;
var b =i++  +  5;
//i++参与运算时，先参与运算，然后自身再+1，故代码可以改写为：b=i+5;i=i+1;
console.log(b); //6
console.log(i); //2

++i

var i = 1;
var b = ++i + 5;
console.log(b); //7
console.log(i)  // 2
//++i参与运算时，先自身再+1，然后再参与运算，故代码可以改写为：i=i+1;b=i+1+5;

eg:

var    a = 10,
    b = 20,
    c = 30;
//不参与运算二者效果是一样的，都是+1
++a; //11
a++; //12
e = ++a + (++b) + (c++) + a++;
13 +    21   +   30  + 13
//     ++a参与运算，故先+1再运算，
//    ++b参与运算，故先+1再运算
//    c++参与运算，先参与运算，自身再+1；
//     a++参与运算，先参与运算，自身再+1，前面++a已经改变了a的值为13
console.log(e);   //77

短路( && 和 || )

  • 短路运算运用在化简条件判断
  • 短路是指前面能得出结果，后面的就不执行，如果前面都不能决定最终结果，结果就是最后一个值
  • 用于形参多于实参

&& 与 和

// && 多条件只要有一个是false，最终结果就是false,如果前面都不能决定最终结果，结果就是最后一个值

console.log(false && true && true); //false
console.log(1 && 0 && NaN); //0

|| 或

// || 多条件只要有一个是true，最终的结果就是true，如果前面都不能决定最终结果，结果就是最后一个值

console.log(true || false || false); //true
console.log(6 || null || ""); //6

function f2(callback) {

     */

    callback && callback();
}

循环结束语句

break

  • 结束后面循环的执行
  • break后面的代码不执行

eg:

// 在循环打印1到5的过程中，打印到3就停下来，不打印了，后面的4，和5也不打印了。

    // 注意要先输出3再结束循环break
    // break后面的代码不执行
    console.log(i);
    if (i == 3) {
        break;
    }
}

循环语句里的break与switch语句里的break区别

循环语句的break属于循环语句,这里的作用是跳出循环

switch语句的break属于switch语句打断后面判断的执行

跳过本次循环

continue关键字可以跳过本次循环，本次循环continue后面的代码不执行，后面的循环会继续执行

eg:

// 在循环打印1到5的过程中，打印到3就停下来，但是后面的4和5还会打印。
// 1 2   4 5
for (var i = 1; i <= 5; i++) {
    if (i == 3) {
        // continue关键字可以跳过本次循环，本次循环continue后面的代码不执行，后面的循环继续执行

        // replace()方法没有改变原字符，切记要重新赋值
        arr[i] = arr[i].replace(arr[i][0], arr[i][0].toUpperCase());
        //console.log(arr[i]);
    }
    str = arr.join("");
    return str;
}

console.log(tuofeng("get-element-by-id"));
console.log(tuofeng("my-name-is-miki"));

函数的参数

函数封装的时候 小括号里面可以书写形参 形参的本质是一个变量，函数体里面也是写形参不要把实参写上去

function getSum(a, b, c) {
        console.log(a + b + c);
}

调用函数的时候，小括号里面传递的数据就是 实参 相当于给变量赋值

getSum(1, 2, 3);

形参和实参根据顺序一一对应

return 关键字

  • return 关键字的作用就是返回函数的执行结果，不返回，默认是返回undefined

  • 函数执行结果返回到函数调用的地方

  • return 返回函数的执行结果并且终止函数的执行，return后面的代码不执行

  • return 只能返回一个结果，就是最后那个为准

  • 对象的用.方法，数组的就用[]

function getSum(a, b, c) {
	var d = a + b + c;
    return d;
    alert("我是return后面的代码");
    // return后面的代码不执行，所以不会弹出
}

eg:

//对象的用点方法
function calc(num1, num2) {
    // return只能返回一个结果，就是最后那个为准
    // return num1 + num2, num1 * num2;
    return { sum: num1 + num2, ji: num1 * num2 }
 }
console.log(calc(10, 20).sum)
console.log(calc(10, 20).ji)
//数组的就用[]
function calc(num1, num2) {
    // return只能返回一个结果，就是最后那个为准
    // return num1 + num2, num1 * num2;
    var arr = [num1 + num2, num1 * num2]
    return arr;
 }

函数封装步骤

  1. 先按照正常业务逻辑书写功能代码

  2. 把这些功能代码放在function 函数名(){功能代码} 函数体内部

  3. 观察那些值是变化的,封装成形参

  4. 有返回结果的就return

  5. 调用的时候传递实参

eg:

// 求n - m之间所有数的和(先写一个求10到100之间的累加和)

function getSum(n, m) {
    // 保存总和
    var sum = 0;
    for (var i = n; i <= m; i++) {
        sum += i;
    }
    return sum;
}
console.log("10到20的整数累加和是" + getSum(10, 20));

求两个数组中的最大值

  • 先书写求最大值的代码

  • 假设第一个是最大，遍历这个数组，用数组里面的每一项去和最大值比较

  • 如果大于这个最大值，就更新这个最大值为比它大的那个数据

    注意：数组也能传进函数当参数；

eg:

var arr1 = [10, 20, 30, 40, 50];
var arr2 = [55, 66, 77, 99, 88, 10];
var max = arr[0];
function getMax(arr) {
    for (var i = 0; i < arr.length; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

// getMax(arr1); // arr1的最大值 数组也可以作为参数去传递 
// getMax(arr2); // arr2的最大值 数组也可以作为参数去传递

var zuida = getMax(arr1) > getMax(arr1) ? getMax(arr1) : getMax(arr2);

alert("两个数组中的最大值是" + zuida);

函数的作用域

  • 作用域scope 产生作用的区域范围
  • 变量产生作用的范围
  • 没有书写在函数内部的变量，是全局变量 作用范围在script标签内部和外链的js都可以访问**(注意执行顺序，先声明后使用)**
  • 书写在函数内部的变量，叫局部变量 函数外面访问不了，作用范围在本函数内部
var a = 10;
function fn() {
    console.log(a); //10
}
fn();

结论：

  • 局部变量，只有函数内部才能使用
  • 全局变量在函数里面也可以访问

函数发生嵌套，就会产生作用域链

作用域链的访问原则：

优先使用本函数内部的变量，本函数里面没有就往上一层函数查找，就近原则，查找到全局作用域都没有找到这个变量，就报错

注意：函数一定要调用才会执行

预解析

预解析是浏览器的一个功能，它会把var声明变量和function声明的函数，提前到当前作用域的最前面，但是并没有赋值

注意 函数调用和赋值的位置不能修改！！！！！！

console.log(a);
var a = 10;

//预解析成功
var a;
console.log(a); //undefined
a = 10;
var num = 10;
fun();
function fun() {
    console.log(num);
    var num = 20;
}

// 预解析成
function fun() {
    var num;
    console.log(num);   //undefined
    num = 20;
}
var num;
num = 10;
fun();

回调函数

函数调用的时候，传递另外一个函数作为实参，这个函数称为 回调函数

函数的参数（关键字：arguments）

  • 在函数里面，有一个关键字 arguments 可以接受传递过来的实参伪数组

  • 伪数组有数组的一些特征(下标个数等)，但是没有数组的一些方法

  • 用于实参比形参多

eg:

function getSum() {
    // 保存总和
    var sum = 0;
    for (var i = 0; i < arguments.length; i++) {
        sum += arguments[i];
    }
    return sum;
}
console.log(getSum(22, 33, 66));
console.log(getSum(1, 2));
