求最大值

思路：

  • 先书写求最大值的代码
  • 假设第一个是最大，遍历这个数组，用数组里面的每一项去和最大值比较
  • 如果大于这个最大值，就更新这个最大值为比它大的那个数据

eg

var arr1 = [10, 20, 30, 40, 50];
var arr2 = [55, 66, 77, 99, 88, 10];
var max = arr[0];
function getMax(arr) {
    for (var i = 0; i < arr.length; i++) {
        if (arr[i] > max) {
            max = arr[i];
        }
    }
    return max;
}

// getMax(arr1); // arr1的最大值 数组也可以作为参数去传递 
// getMax(arr2); // arr2的最大值 数组也可以作为参数去传递

var zuida = getMax(arr1) > getMax(arr1) ? getMax(arr1) : getMax(arr2);

alert("两个数组中的最大值是" + zuida);

eg1

var btn = document.querySelectorAll("button");

eg2

排他的其他写法(事件委托的排他)

var box = document.querySelector("#box");
// 创建一个变量来存储上一个点击的元素
var prev = null;
box.onclick = function (e) {
    // 如果点击的目标元素不是box这个对象
    if (e.target !== box) {
        // 如果存在上一次点击的元素
        if (prev) {
            // 就去掉背景色
            prev.style.background = "";
        }
        // 再给目标元素添加背景色
        e.target.style.background = "tan";
        // 添加完之后把这个目标元素设置为上一次点击的元素
        prev = e.target;
    }
}

出现次数最多 - 出现了多少次

常见于字符串，数组中

思路：

  1. 先把数组转化成字符串，str[下标]可以得到每一个字
  2. 新建一个空的对象来储存每一个字的属性，和每个字出现次数的值
  3. 判断新建的这个对象里面，有没有每个字str[i]
  4. 如果没有就设置这个属性的值为出现了一次，如果本来就有就+1次 ++

eg

var students = [...,...,...,];
//将数组转化成字符串
var str = students.join("");
//声明一个空对象
var obj = {};

for (var i = 0; i < str.length; i++) {
    // char是每个字符
    // obj[char]是这个对象有没有这个字符属性
    var char = str[i];
    if (obj[char]) {
        // 有这个属性就加一次
        obj[char]++;
    } else {
        // 没有就把这个属性赋值为一次
        obj[char] = 1;
    }
}
// 输出这个保存了每个字和出现次数的对象
console.log(obj);

// 先设置一个最大值的初始值
var max = 0;
// 新建一个变量来储存出现最多次数的字符
var maxchar = "";

for (var key in obj) {
    if (obj[key] > max) {
        max = obj[key];
        maxchar = key;
    }
}
console.log(`出现次数最多的字符是${maxchar},出现的次数数${max}次`);

三维焦点图(轮播)

思路:

  1. 取数据
  2. 把每个li的类名提出来，存进一个新数组里面；
  3. 点击下一张，就是把新数组里面 把数组里面的第一个shift，放到数组的最后push
  4. 点击上一张，就是把新数组里面 把数组里面的最后一个pop，放到数组的前面unshift
  5. 以上两个步骤都要将新数组里面的类，重新赋值给li

eg

// 取数据
var lis = document.querySelectorAll(".banner>ul>li");
var prev = document.querySelector(".prev");
var next = document.querySelector(".next");

// 专门新建一个数组来存类名
var aclass = [];
console.log(lis, prev, next);

for (var i = 0; i < lis.length; i++) {
    // console.log(lis[i].className);
    // 把每一个li的类名放到aclass数组里面
    aclass.push(lis[i].className);
}
// 显示类名
console.log(aclass);

// 点击下一张
next.onclick = function () {
    // 把数组里面的第一个shift，放到数组的最后push
    aclass.push(aclass.shift());
    console.log(aclass);
    // 给li重新修改类名  取相同的下标但是不同的类名
    for (var i = 0; i < lis.length; i++) {
        lis[i].className = aclass[i];
    }
}

// 点击上一张
prev.onclick = function () {
    // 把数组里面的最后一个pop，放到数组的前面unshift
    aclass.unshift(aclass.pop());
    console.log(aclass);
    // 给li重新修改类名  取相同的下标但是不同的类名
    for (var i = 0; i < lis.length; i++) {
        lis[i].className = aclass[i];
    }

}

反选

思路：

  1. 取数组；
  2. 点击全选选框，设置下面全部的选框全部选中，并且把全选的文字变成全不选
  3. 点击下面的每一个选框的时候，都要全部检查一遍，有没有没选中的，如果有就不能全选，如果都选中了，就可以全选
  4. 新建一个变量来控制是全选 true 全选 false不全选**(关键：)**
  5. 点击反选的按钮，设置全部的选框选中状态取反(顺便控制全选 要封装函数)
  6. 新建一个变量来控制是全选 true 全选 false不全选**(关键：)**

eg

点击生成表格

输入行数：<input type="text" id="row">
<br>
输入列数：<input type="text" id="col">
<br>
<button>生成表格</button>
  • 点击按钮，生成rows行tr，并且把这些行插入到table对象里面
  • 点击按钮，生成cols列td，并且把这些列插入到tr对象里面
  • 外层循环tr**(有多少行)**
  • 里层循环td**(有多少列)**
  • 注意：请思考题目是否需要非空校验
var btn = document.querySelector("button");
var row = document.getElementById("row");
var col = document.getElementById("col");

btn.onclick = function () {
    // 非空校验
    if (row.value != "" && col.value != "") {
        var rows = +row.value;  //行数
        var cols = +col.value;  //列数
        var table = document.createElement("table");
        // 外层循环控制行数tr
        for (var i = 1; i <= rows; i++) {
            var tr = document.createElement("tr");
            // 里层循环控制列数td
            for (var j = 1; j <= cols; j++) {
                var td = document.createElement("td");
                td.innerHTML = `第${i}行,第${j}列`;
                // 把td插入tr里面
                tr.appendChild(td);
            }
            // 把tr插入table后面
            table.appendChild(tr);
        }
        // 把表格插入body后面
        document.body.appendChild(table);
        // 清空输入框的值
        row.value = "";
        col.value = "";
    } else {
        alert("请输入内容")
    }
}

function getRandomInclusive(min, max) {
    min = Math.ceil(min);
    max = Math.floor(max);
    return Math.floor(Math.random() * (max - min + 1)) + min;   //含最大值，含最小值
}

var arr = [];
for (var i = 1; i <= 10; i++) {
    var ranNum = getRandomInclusive(10, 100);
    arr.push(ranNum);
}

var students = ["..","..","..","..","..","..","..",".."];

var list = document.querySelector(".list");
//根据数组的个数生成多个li
for (var i = 0; i < students.length; i++) {
    // 创建多个li
    var lis = document.createElement("li");
    // 设置li里面的文字是数组的每一项
    lis.innerText = students[i];
    // 把li放到ul里面
    list.appendChild(lis);
}
// 点击 开始 按钮的时候
// 就随机从 0 ~ students.length; 生成这个范围的随机数
var btn = document.querySelector("button");
var lis = document.querySelectorAll(".list>li");
var timer = null; //先设置成空对象

btn.onclick = function () {
    // 如果 按钮 里面的文字是开始 就 变成暂停 顺便开启定时间器 
    // 否则清除定时器，并且把文字变成暂停
    if (btn.innerText == "开始") {
        timer = setInterval(function () {
            var ranNum = parseInt(Math.random() * students.length);
            // 让定时器不断的随机同学
            console.log(students[ranNum]);
            // 设置全部清除类名（排他）
            for (var j = 0; j < lis.length; j++) {
                lis[j].classList.remove("cur");

            }
            // 再给随机到的这个增加类名（确立自己）
            lis[ranNum].classList.add("cur");

        }, 20);
        // 把按钮的文字变成暂停
        btn.innerText = "暂停";

    } else {
        //清除定时器
        clearInterval(timer);
        // 把按钮的文字变成开始
        btn.innerText = "开始";
    }
}
