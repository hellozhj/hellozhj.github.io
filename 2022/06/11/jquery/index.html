排他的其他写法(事件委托的排他)

tips：创建一个变量来储存上一个点击的元素

var box = document.querySelector("#box");
// 创建一个变量来存储上一个点击的元素
var prev = null;
box.onclick = function (e) {
    // 如果点击的目标元素不是box这个对象
    if (e.target !== box) {
        // 如果存在上一次点击的元素
        if (prev) {
            // 就去掉背景色
            prev.style.background = "";
        }
        // 再给目标元素添加背景色
        e.target.style.background = "tan";
       // 添加完之后把这个目标元素设置为上一次点击的元素
        prev = e.target;
    }
}

需求：

  • 鼠标移上li的时候，隐藏p 显示img
  • 鼠标离开li的时候，显示p 隐藏img

原生

var lis = document.querySelectorAll("ul>li");
for (var i = 0; i < lis.length; i++) {
    lis[i].onmouseenter = function () {
        this.children[0].style.display = "none";
        this.children[1].style.display = "block";

        this.children[0].style.display = "block";
        this.children[1].style.display = "none";

}

事件委托

  • 事件委托的作用：

  • 性能要好（为多个相同元素绑定事件）

  • 针对新创建的元素，直接可以拥有事件（为不存在的元素绑定事件）

  • jQuery事件委托语法

    $("选择器父级").on("事件类型", "后代元素", function () {  });

eg:

$("button:eq(0)").click(fabu);

function fabu() {
    // 表单的内容去掉前后空格
    var val = $.trim($("input").val());
    // 如果不是空的
    if (val !== "") {
        // 就创建一个li
        var li = `<li>${val}<button>x</button></li>`;
        // 添加到ul里面的前面
        $("ul").prepend(li);
        $("input").val("");
    } else {
        alert("请输入内容");
    }
}
// 网页文档document的任意位置按回车键位
$(document).keyup(function (e) {
    // console.log(e.keyCode)
    if (e.keyCode == 13) {
        fabu();
    }
})


// 注意 jQ事件委托里面的$(this)是点击的目标元素 和 原生 js不一样

// 需求：将旅游日记列表的标题渲染到页面显示
// 1：请求数据
$.ajax({
    url: "http://kumanxuan1.f3322.net:8809/travels/query",
    success(res) {
        console.log(res);
        // 2:做一个响应处理
        if (res.code == 200) {
            // 业务逻辑(给原本的数据进行一个清空处理)
            $("ul").empty();
            // 3循环添加(DOM渲染)
